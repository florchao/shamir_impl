# Compiler
CC := gcc

# Directories
SRCDIR := src
INCDIR := include
BUILDDIR := build
TARGET := ss

# Source files
SRCEXT := c
SOURCES := $(shell find $(SRCDIR) -type f -name *.$(SRCEXT))

# Object files
OBJECTS := $(patsubst $(SRCDIR)/%,$(BUILDDIR)/%,$(SOURCES:.$(SRCEXT)=.o))

# Flags
CFLAGS := -g -Wall -I$(INCDIR)

# Linker flags
LDFLAGS := -lm

# Target
$(TARGET): $(OBJECTS)
	@mkdir -p $(@D)
	$(CC) $^ -o $(TARGET) $(LDFLAGS)

# Object files
$(BUILDDIR)/%.o: $(SRCDIR)/%.$(SRCEXT)
	@mkdir -p $(@D)
	$(CC) $(CFLAGS) -c -o $@ $<

.PHONY: clean

# Clean
clean:
	@echo " Cleaning..."
	$(RM) -r $(BUILDDIR) $(TARGET)
